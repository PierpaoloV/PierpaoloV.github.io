OWNER = "PierpaoloV"
REPO = "learning"
RAW = `https://raw.githubusercontent.com/${OWNER}/${REPO}/main`
API = `https://api.github.com/repos/${OWNER}/${REPO}`

// Fetch full file tree once to know what exists
treeJson = fetch(`${API}/git/trees/main?recursive=1`).then(r => r.json())
existingPaths = new Set((treeJson.tree || []).map(f => f.path))

async function fetchRaw(path) {
  try {
    const r = await fetch(`${RAW}/${path}`)
    if (!r.ok) return null
    return await r.text()
  } catch { return null }
}

function isSolved(content) {
  if (!content) return false
  const lines = content.split('\n').filter(l => l.trim() && !l.trim().startsWith('#') && !l.startsWith('from ') && !l.startsWith('import ') && !l.startsWith('class ') && !l.includes('Optional') && !l.includes('List'))
  const body = lines.join('').replace(/\s/g, '')
  return body !== '' && !body.endsWith('pass') && !body.includes('Writeyoursolutionhere')
}

function isWritten(content) {
  if (!content) return false
  return content.trim().length > 80 && !content.includes('Write your design here') && !content.includes('Write your solution here')
}

// Determine current week (0-indexed)
currentWeekIdx = {
  const today = new Date()
  const starts = schedule.weekStarts.map(s => new Date(s))
  let idx = 0
  for (let i = 0; i < starts.length; i++) {
    if (today >= starts[i]) idx = i
  }
  return idx
}

// Fetch all solution + review statuses in parallel
allProblems = schedule.weeks.flatMap(w => w.problems)

problemStatuses = Promise.all(
  allProblems.map(async p => {
    const paths = await existingPaths
    const content = await fetchRaw(`leetcode/${p.slug}/solution.py`)
    const solved = isSolved(content)
    const reviewed = paths.has(`leetcode/${p.slug}/review.md`)
    const reviewContent = reviewed ? await fetchRaw(`leetcode/${p.slug}/review.md`) : null
    return { slug: p.slug, solved, reviewed, reviewContent }
  })
)

statusMap = Object.fromEntries(problemStatuses.map(s => [s.slug, s]))

// Fetch system design statuses
allDesigns = schedule.weeks.filter(w => w.systemDesign).map(w => w.systemDesign)

designStatuses = Promise.all(
  allDesigns.map(async d => {
    const content = await fetchRaw(`system-design/${d.slug}/design.md`)
    return { slug: d.slug, written: isWritten(content) }
  })
)

designMap = Object.fromEntries(designStatuses.map(d => [d.slug, d]))

// Progress calculations
weekProgress = {
  const sm = await statusMap
  const week = schedule.weeks[currentWeekIdx]
  const total = week.problems.length
  const solved = week.problems.filter(p => sm[p.slug]?.solved).length
  return { solved, total, pct: total > 0 ? Math.round(solved / total * 100) : 0 }
}

monthProgress = {
  const sm = await statusMap
  const total = allProblems.length
  const solved = allProblems.filter(p => sm[p.slug]?.solved).length
  return { solved, total, pct: total > 0 ? Math.round(solved / total * 100) : 0 }
}
function diffBadge(d) {
  const colors = { Easy: "#22c55e", Medium: "#f59e0b", Hard: "#ef4444" }
  return `<span style="background:${colors[d]||'#6b7280'};color:white;padding:1px 8px;border-radius:99px;font-size:0.72rem;font-weight:700;">${d}</span>`
}

function statusIcon(ok) {
  return ok ? `<span style="color:#22c55e;font-size:1rem;">✅</span>`
             : `<span style="color:#d1d5db;font-size:1rem;">⬜</span>`
}

function bar(pct, label, sub) {
  return `
  <div style="margin-bottom:1.1rem;">
    <div style="display:flex;justify-content:space-between;margin-bottom:5px;">
      <span style="font-weight:600;font-size:0.9rem;">${label}</span>
      <span style="color:#6b7280;font-size:0.85rem;">${sub}</span>
    </div>
    <div style="background:#e5e7eb;border-radius:999px;height:13px;overflow:hidden;">
      <div style="background:linear-gradient(90deg,#2E8B7A,#52b6a8);width:${pct}%;height:100%;border-radius:999px;transition:width 0.6s ease;min-width:${pct>0?'6px':'0'};"></div>
    </div>
  </div>`
}

function reviewBlock(content) {
  if (!content) return ''
  const preview = content.slice(0, 1800) + (content.length > 1800 ? '\n...' : '')
  return `
  <details style="margin-top:6px;">
    <summary style="cursor:pointer;color:#2E8B7A;font-size:0.82rem;font-weight:500;user-select:none;">📝 Grader feedback</summary>
    <div style="background:#f8fafc;border-left:3px solid #2E8B7A;padding:0.75rem 1rem;margin-top:6px;font-size:0.78rem;white-space:pre-wrap;font-family:'Fira Mono',monospace;line-height:1.5;border-radius:0 6px 6px 0;overflow-x:auto;">${preview.replace(/</g,'&lt;').replace(/>/g,'&gt;')}</div>
  </details>`
}
